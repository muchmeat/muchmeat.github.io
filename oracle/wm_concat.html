# WM_CONCAT

在处理超过4000字符时就会出错，可用可以函数 RX_CONCAT 替代

--创建一个动态数组
CREATE OR REPLACE TYPE RXARRAY is TABLE OF VARCHAR2(4000);

create or replace type rx_wm_concat authid current_user as object
(
 
--拼接字符串,存放中间值，当然也可以定义为clob，clob会使用临时段，导致临时表空间迅速增大;
--查看wmsys下的function可以发现Oracle10g到oracle11g自带的wm_concat函数的返回类型从clob变成varchar2
  currStr VARCHAR2(4000),
 
-- 定义字符串数组
  strArray RXARRAY,
 
-- 初始化接口函数
  STATIC FUNCTION ODCIAGGREGATEINITIALIZE(init IN OUT rx_wm_concat)
    RETURN NUMBER,
 
-- 迭代接口函数
  MEMBER FUNCTION ODCIAGGREGATEITERATE(self     IN OUT rx_wm_concat,
                                       colValue IN VARCHAR2) RETURN NUMBER,
 
-- 并行时字符串合并的接口函数
  MEMBER FUNCTION ODCIAGGREGATEMERGE(self IN OUT rx_wm_concat,
                                     next rx_wm_concat) RETURN NUMBER,
 
-- oracle终止接口函数
  MEMBER FUNCTION ODCIAGGREGATETERMINATE(self        IN rx_wm_concat,
                                         returnValue OUT VARCHAR2,
                                         flags       IN NUMBER)
    RETURN NUMBER
);

create or replace type body rx_wm_concat is
  --定义函数的body
 
  --初始化函数
  STATIC FUNCTION ODCIAGGREGATEINITIALIZE(init IN OUT rx_wm_concat)
    RETURN NUMBER is
begin
    init := rx_wm_concat('', RXARRAY());
return ODCICONST.Success;
END;
 
  -- 字符串拼接，self 为当前聚集函数的指针,用来与前面的计算结果进行关联
  MEMBER FUNCTION ODCIAGGREGATEITERATE(self     IN OUT rx_wm_concat,
                                       colValue IN VARCHAR2) RETURN NUMBER is
    temp_tag char(1) default '0';
begin
for i in 1 .. self.strArray.count loop
      if colValue = self.strArray(i) then
        temp_tag := '1';
        exit;
end if;
end loop;
   
    if temp_tag != '1' and colValue is not null then
     
      -- 存放入数组
      self.strArray.extend;
     
      self.strArray(self.strArray.count) := colValue;
end if;

return ODCICONST.Success;
END;
 
  --并行操作是用来合并两个聚集函数的两个不同的指针对应的结果
  MEMBER FUNCTION ODCIAGGREGATEMERGE(self IN OUT rx_wm_concat,
                                     next rx_wm_concat) RETURN NUMBER is
    temp_tag char(1) default '0';
begin
   
    -- 将next数组中元素全部放入self指针对应的数组中
for i in 1 .. next.strArray.count loop
      for j in 1 .. self.strArray.count loop
        if next.strArray(i) = self.strArray(j) then
          temp_tag := '1';
          exit;
end if;
end loop;
      if temp_tag != '1' and next.strArray(i) is not null then
        -- 存放入数组
        self.strArray.extend;
       
        self.strArray(self.strArray.count) := next.strArray(i);
end if;
      temp_tag := '0';
end loop;

return ODCICONST.Success;
END;
 
  -- 终止函数，返回结果
  MEMBER FUNCTION ODCIAGGREGATETERMINATE(self        IN rx_wm_concat,
                                         returnValue OUT VARCHAR2,
                                         flags       IN NUMBER) RETURN NUMBER IS
    temp_rtnValue varchar2(4000);

BEGIN
for i in 1 .. self.strArray.count loop
      temp_rtnValue := temp_rtnValue || ',' || self.strArray(i);
end loop;
    returnValue := ltrim(temp_rtnValue, ',');

return ODCICONST.Success;
END;
END;

create or replace function rx_concat(colValue  VARCHAR2) RETURN VARCHAR2
  AGGREGATE USING rx_wm_concat;
