# 一些SQL

# 分割字符串拿出最后一个元素

SELECT SUBSTR(column_name, INSTR(column_name, '/', -1) + 1) AS last_element
FROM table_name;

# 1970年至今毫秒数转为date

注意东八区时差问题

SELECT TO_DATE('1970-01-01', 'YYYY-MM-DD') + (1 / 24 / 60 / 60 / 1000) * <milliseconds> AS converted_date
FROM dual;

SELECT TO_DATE('1970-01-01 08:00:00', 'YYYY-MM-DD HH24:MI:SS') + <milliseconds> / 86400000 AS converted_date
FROM dual;

# 查询锁定记录并删除

SELECT s.sid, s.serial# FROM v$locked_object lo, dba_objects ao, v$session s WHERE ao.object_id = lo.object_id AND lo.session_id = s.sid;

ALTER system KILL session 'SID,serial#';
