# ORA-01841

  • 日期字符串中包含了非法的字符
--1.创建异常记录表
create table date_EXCEPTION
(
  id   INTEGER not null,
  timestamp  VARCHAR2(50)
);

--2.执行数据筛查
DECLARE
  V_DATE_STRING date_timestamp_test.timestamp % TYPE;
  TEMP          DATE;
BEGIN
  FOR REC IN (SELECT ID, timestamp FROM date_timestamp_test) LOOP
    V_DATE_STRING := REC.timestamp;
    BEGIN
      SELECT TO_DATE('1970-01-01 08:00:00', 'YYYY-MM-DD HH24:MI:SS') +
             V_DATE_STRING / 86400000
        INTO TEMP
        from dual;
    exception
      when others then
        insert into date_EXCEPTION values (rec.id, V_DATE_STRING);
    end;
  end loop;
end;

--3.查看异常数据
select * from date_EXCEPTION;
