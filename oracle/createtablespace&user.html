# 使用sys dba 创建表空间及用户

# 查看DBF文件位置

select tablespace_name, file_name
from dba_data_files;

# 创建表空间

create
tablespace zhzhpt_data datafile 'D:\app\user\oradata\orcl\zhzhpt_data.dbf' size 100m autoextend on next 10m;

# 创建临时表空间

create
temporary
tablespace zhzhpt_data_temp tempfile 'D:\app\user\oradata\orcl\zhzhpt_data_temp.dbf' size 100m autoextend on next 10m;

# 创建用户并设置默认表空间

create
user zhzhpt identified by zhzhpt default tablespace zhzhpt_data;

create
user zhzhpt identified by zhzhpt default tablespace zhzhpt_data temporary tablespace zhzhpt_data_temp;

# 用户授权

grant connect,resource to zhzhpt;

--dba权限可以用于cmd导入dmp文件时
grant dba to zhzhpt;
