# ubuntu安装mysql8.0.26

以下内容环境均来自于windows10，ubuntu20.04版本

参考 https://www.ngui.cc/el/2433772.html?action=onClick

# 列出可用的软件版本

apt-cache madison 软件名称

如软件名称为 mysql-server

# 指定版本安装

sudo apt install 软件名称=版本号

# 查看MySQL版本

mysqladmin --version

# 验证是否安装mysql

dpkg -l | grep mysql

# 初始化配置信息

mysql_secure_installation

# 查看MySQL服务启动状态

其中 mysql 是服务名

service mysql status

# 启动MySQL

其中 mysql 是服务名

service mysql start

# 登录mysql

mysql -u root -p

以下方式不太顺利，没有成功。最好使用root用户进行操作

# 下载deb

https://downloads.mysql.com/archives/community/

选择对应的版本

# 解压deb

将 tar文件 解压到 统计目录下 mysql文件夹

sudo tar -xvf mysql-server_8.0.26-1ubuntu20.04_amd64.deb-bundle.tar -C mysql

# 安装deb

以下内容按照顺序执行，可能会有偏差

sudo dpkg -i libmysqlclient21_8.0.26-1ubuntu20.04_amd64.deb
sudo apt install libmecab2
sudo dpkg -i libmysqlclient-dev_8.0.26-1ubuntu20.04_amd64.deb
sudo dpkg -i mysql-community-server-core_8.0.26-1ubuntu20.04_amd64.deb
sudo dpkg -i mysql-community-server-debug_8.0.26-1ubuntu20.04_amd64.deb
sudo dpkg -i mysql-community-server_8.0.26-1ubuntu20.04_amd64.deb
sudo dpkg -i mysql-community-client-core_8.0.26-1ubuntu20.04_amd64.deb
sudo dpkg -i mysql-community-client-plugins_8.0.26-1ubuntu20.04_amd64.deb
sudo dpkg -i mysql-community-client_8.0.26-1ubuntu20.04_amd64.deb
