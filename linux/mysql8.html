# Linux安装mysql

# 查看glibc版本

rpm -qa | grep glibc

# 下载安装

下载地址 (opens new window)

# 上传文件到Linux

使用finalshell链接Linux服务器，通过上传功能将下载的文件上传到Linux

可以先创建一个 mysql 文件夹，将文件放入该文件夹

mkdir mysql

# 解压MySQL压缩包

查看自己下载的文件名，执行类似以下命令

tar xvJf mysql-8.2.0-linux-glibc2.17-x86_64.tar.xz

# 重命名解压后的MySQL

解压之后的MySQL文件夹名称太长了，可以 mv “原文件名称” “新文件名称”，重命名一下，注意如果是在当前文件中重命名，新文件名中不能包含斜杆“/”

mv mysql-8.2.0-linux-glibc2.17-x86_64 mysql-8.2.0

# 进入MySQL解压后根目录，创建data文件夹

cd mysql-8.2.0
mkdir data

# 创建mysql组和用户

# 创建mysql组

groupadd mysql

# 创建mysql用户

useradd -g mysql mysql

证明一下创建成功了

# 将安装的MySQL权限给mysql用户

命令格式 chown -R “mysql组名”.“mysql用户名” “mysql安装根目录”

chown -R mysql.mysql /usr/local/mysql/mysql-8.2.0

注意大小写

# MySQL初始化

Linux中的MySQL默认是区分表名大小写的，如果你是安装的也是MySQL8，那么不区分大小写需要在数据库初始化时设置，不然初始化后，在配置文件在设置重启时是会报错的

进入mysql的bin文件夹

cd bin/

./mysqld --user=mysql --basedir=/usr/local/mysql/mysql-8.2.0 --datadir=/usr/local/mysql/mysql-8.2.0/data --lower-case-table-names=1 --initialize

出现了以下问题，这是一个版本兼容性问题，我安装的mysql版本比当前的Linux系统依赖版本高。

./mysqld: /lib64/libstdc++.so.6: version `CXXABI_1.3.11' not found (required by ./mysqld)

不想对当前系统产生什么影响，决定下载适应当前系统的版本。按照之前的步骤再来一遍

这时再执行上面的命令出现

# 启动Mysql

service mysql start
service mysql restart

# Mysql状态

service mysql status
