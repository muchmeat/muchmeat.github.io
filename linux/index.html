# linux常用命令

可能有错误

# 查看是否开启转发

1表示已开启，0表示未开启

sudo sysctl net.ipv4.ip_forward

# 创建文件夹

sudo mkdir 文件夹

# 创建文件

sudo touch 文件名

# 删除文件

sudo rm 文件名

# 删除文件夹

sudo rmdir 空文件夹
sudo rm -rf 非空文件夹

# 查看进程

# mysql
ps -ef|grep mysqld

# 杀掉进程

kill -9 进程号

# 查看端口

netstat -ano |grep :6379
