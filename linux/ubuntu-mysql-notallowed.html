以下内容环境均来自于windows10，ubuntu20.04版本

参考 https://blog.csdn.net/weixin_42599091/article/details/125224850

# 登录数据库

mysql -u root -p

# 进入mysql数据库查看root信息

use mysql;
select host from user where user='root';

# 修改host值

将host值localhost修改为%

update user set host = '%' where user ='root';

# 刷新权限配置

flush privileges;
