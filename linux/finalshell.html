# 配置Ubuntu中ssh

可能用到的命令

--安装ssh
sudo apt-get install openssh-server
sudo apt-get install openssh-client
--启动ssh
sudo service ssh start
--重启ssh
sudo service ssh restart
--查看ssh服务是否启动
sudo ps -e |grep ssh
--生成ssh key
sudo ssh-keygen -A
--查看防火墙状态
sudo ufw status
--关闭防火墙
sudo ufw disable
--开放22端口 给finalshell
sudo ufw allow 22
--允许ssh
sudo ufw allow ssh
--系統更新 不要使用！！！！！
sudo apt update && sudo apt upgrade


--编辑ssh配置
sudo vi /etc/sshd/config

finalshell不停提示输入密码 (opens new window) 設置如下

PermitRootLogin yes
