# 请求传参

# 1.普通对象

# 1.1.JSON请求

请求类似 其中参数

{
  "kssj": "1",
  "jssj": "2"
}

接收方式1

@RequestMapping("testPost")
@ResponseBody
public JSONResponse testPost(@RequestBody Map map) {
    return JSONResponse.OK();
}

接收方式2

@RequestMapping("testPost2")
@ResponseBody
public JSONResponse testPost(@RequestBody PostParams params) {
    return JSONResponse.OK();
}

class PostParams{
    String kssj;
    String jssj;
    //getter、setter
}

# 1.2.非JSON请求

请求类似

接收方式1

@RequestMapping("testPost3")
@ResponseBody
public JSONResponse testPost3(String kssj,String jssj) {
    log.info("kssj{}",kssj);
    log.info("jssj{}",jssj);
    return JSONResponse.OK();
}

接收方式2

这里需要注意 @RequestParam 注解的用法

@RequestMapping("testPost4")
@ResponseBody
public JSONResponse testPost4(@RequestParam String kssj,@RequestParam(required = false) String jssj) {
    return JSONResponse.OK();
}

接收方式3

可直接使用对象接收

@RequestMapping("testPost5")
@ResponseBody
public JSONResponse testPost5(PostParams postParams) {
    return JSONResponse.OK();
}
