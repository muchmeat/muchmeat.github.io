# 多集合合并为一个集合并且去除重复元素

核心：需要在元素对象内重写equals及hashCode方法

1.集合内对象

import java.util.Objects;

public class SysUser {

    private Long id;
    private String name;

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        SysUser sysUser = (SysUser) o;
        return Objects.equals(id, sysUser.id);
    }

    @Override
    public int hashCode() {
        return Objects.hash(id);
    }
}

2.实现合并

import java.util.ArrayList;
import java.util.List;
import java.util.stream.Collectors;
import java.util.stream.Stream;

public class MultiList2One {

    public static void main(String[] args) {
        List<SysUser> list1 = new ArrayList<>();
        for (long i = 0; i < 3L; i++) {
            SysUser sysUser = new SysUser();
            sysUser.setId(i + 1);
            sysUser.setName("用户" + (i + 1));
            list1.add(sysUser);
        }

        List<SysUser> list2 = new ArrayList<>();
        for (long i = 2; i < 5L; i++) {
            SysUser sysUser = new SysUser();
            sysUser.setId(i + 1);
            sysUser.setName("用户" + (i + 1));
            list2.add(sysUser);
        }

        List<SysUser> collect = Stream.concat(list1.stream(), list2.stream()).distinct().collect(Collectors.toList());
        for (SysUser sysUser : collect) {
            System.out.printf("id:%d,用户：%s", sysUser.getId(), sysUser.getName());
            System.out.println();
        }
    }
}
